class CashMachine
  def initialize
    @balance = 100.0
  end

  def deposit(amount)
    if amount > 0
      @balance += amount
      puts "Внесено #{amount}. Новый баланс: #{@balance}"
    else
      puts "Недопустимая сумма для внесения. Сумма должна быть больше 0."
    end
  end

  def withdraw(amount)
    if amount > 0 && amount <= @balance
      @balance -= amount
      puts "Снято #{amount}. Новый баланс: #{@balance}"
    else
      puts "Недопустимая сумма для снятия. Сумма должна быть больше 0 и не превышать текущий баланс."
    end
  end

  def check_balance
    puts "Текущий баланс: #{@balance}"
  end
end


atm = CashMachine.new
atm.deposit(50)
atm.withdraw(25)
atm.check_balance
