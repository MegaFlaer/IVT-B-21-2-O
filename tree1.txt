def array_x2(array)
  if array.empty?
    puts 'Массив пустой'
    return []
  end

  result = array.map do |element|
    next unless element.is_a?(Integer)

    element * 2
  end.compact

  if result.empty?
    puts 'Невозможно обработать запрос: массив не содержит целочисленные элементы'
  else
    result
  end
end

arr1 = []                       # Пустой массив
arr2 = [1, 2, 3, 'four', 5]     # Массив с различными типами данных
arr3 = ['one', 'two', 'three']  # Массив без целочисленных элементов
arr4 = [1, 2, 3, 4, 5]          # Массив с целочисленными элементами

puts array_x2(arr1)  # Массив пустой

# Массив содержит не только целочисленные элементы
# Выводит только результат с целочисленными элементами: [2, 4, 6, 10]
puts array_x2(arr2)

puts array_x2(arr3)  # Невозможно обработать запрос: массив не содержит целочисленные элементы

# Массив с целочисленными элементами
# Выводит результат: [2, 4, 6, 8, 10]
puts array_x2(arr4)
